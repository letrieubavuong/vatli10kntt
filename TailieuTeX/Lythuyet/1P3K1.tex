\subsubsection{LỰC VÀ BIỂU DIỄN LỰC TÁC DỤNG}
\paragraph{Lực}
\newghichu{\textit{Lực là đại lượng vectơ đặc trưng cho tác dụng của vật này lên vật khác, kết quả làm cho vật thay đổi vận tốc (tức là gây gia tốc cho vật) hoặc làm vật biến dạng}}
\paragraph{Đặc trưng của lực}
\begin{itemize}
\item Các yếu tố của lực bao gồm điểm đặt, phương, chiều và độ lớn.
\item Đường thẳng mang vectơ lực gọi là giá của lực.
\end{itemize}

\subsubsection{TRỌNG LỰC}
\paragraph{Lực hấp dẫn}
\textit{Mọi vật trong vũ trụ đều hút nhau bởi một lực gọi là lực hấp dẫn}
\immini{\paragraph{Trọng lực}
\begin{itemize}
\item Trọng lực là lực hấp dẫn do Trái Đất tác dụng lên vật. Trọng lực là một trường hợp riêng của lực hấp dẫn.
\item Trọng lực được kí hiệu là vectơ $\vec P$, có:
\begin{itemize}
\item Phương thẳng đứng;
\item Chiều hướng về tâm Trái Đất.
\item Điểm đặt của trọng lực gọi là trọng tâm của vật.
\item Độ lớn: $P=m.g$.
\end{itemize}	
\end{itemize}}{\begin{tikzpicture}
\node[] at (0,0){\includegraphics[width=3cm]{images/17-1}};
\end{tikzpicture}}	
\immini{\paragraph{Trọng lượng}
\begin{itemize}
\item Khi vật đứng yên trên Trái Đất, trọng lượng của vật bằng độ lớn của trọng lực tác dụng lên vật:
\begin{align*}
P=m.g
\end{align*}
\item Ở gần mặt đất, gia tốc rơi tự do có giá trị gần đúng $g=9,8\ m/s^2$.

\paragraph{Phân biệt trọng lượng và khối lượng}
\begin{itemize}
\item Trọng lượng của một vật thay đổi khi đem đến một nơi khác có gia tốc rơi tự do thay đổi.
\item Khối lượng là số đo lượng chất của vật. Vì vậy, khối lượng của một vật không thay đổi khi ta chuyển nó từ nơi này đến nơi khác.
\end{itemize}	
\immini{\paragraph{Xác định trọng tâm của một vật phẳng, mỏng}
\begin{itemize}
\item Bằng thực nghiệm ta có thể xác định trọng tâm của một vật phẳng, mỏng có hình dạng bất kì.		 
\item Trọng tâm của các vật phẳng, mỏng và có dạng hình học đối xứng nằm ở tâm đối xứng của vật.
