\paragraph{Lực cản của chất lưu}
\immini{\begin{itemize}
\item Thông thường thuật ngữ chất lưu được dùng để chỉ chất lỏng và chất khí.
\item Mọi vật chuyển động trong chất lưu luôn chịu tác dụng bởi lực cản của chất lưu.
\item Lực này ngược hướng với chuyển động và cản trở chuyển động của vật.
\item Lực cản của chất lưu có tác dụng tương tự như lực ma sát, chúng làm chuyển động của các vật bị chậm lại.
\item Lực cản phụ thuộc vào hình dạng và tốc độ của vật.
\end{itemize}}{\begin{tikzpicture}

\paragraph{Lực nâng của chất lưu}
\begin{itemize}
\item Khi vật chuyển động trong nước hay trong không khí thì ngoài lực cản (của không khí, của nước), vật còn chịu tác dụng của lực nâng.
\end{itemize}

\paragraph{Lực đẩy Archimedes}
\begin{itemize}
\item Lực đẩy Archimedes là trường hợp riêng của lực nâng vật đứng yên trong chất lưu.
\item Công thức tính lực đẩy Archimedes: 
\begin{align*}
\boder{F_A=\rho g V}
\end{align*}
Trong đó:
\begin{itemize}
\item $F_A$: lực đẩy Archimedes (N).
\item $\rho$: Khối lượng riêng của chất lỏng $(kg/m^3)$
\item V: thể tích phần chất lỏng bị chiếm chỗ $(m^3)$
\end{itemize}
\end{itemize}
