\subsubsection{Lực ma sát nghỉ}
\immini{Lực ma sát nghỉ là lực ma sát tác dụng lên mặt tiếp xúc của vật, ngăn không cho vật chuyển động trên một bề mặt. Khi lực tác dụng có độ lớn đạt tới một giá trị nhất định thì vật bắt đầu chuyển động}{\begin{tikzpicture}

\begin{note}
Đặt trên bàn một vật nặng có dạng hình hộp.
\begin{center}
\begin{tikzpicture}
\node[] at (0,0){\includegraphics[width=12cm]{images/18-2}};
\end{tikzpicture}
\end{center}
\begin{itemize}
\item Lúc đầu ta đẩy vật bằng một lực nhỏ, vật không chuyển động (hình a). Lực ma sát nghỉ đã ngăn không cho vật chuyển động.
\item Tăng lực đẩy đến khi lớn hơn một giá trị $F_0$ nào đó (hình b) thì vật bắt đầu trượt. Điều đó chứng tỏ lực ma sát nghỉ có giá trị cực đại $F_0$. Khi lực đẩy (hay kéo) vật $F>F_0$ thì vật bắt đầu trượt.
\item Khi vật đã trượt, ta chỉ cần đẩy vật bằng một lực nhỏ hơn giá trị $F_0$ vẫn duy trì được chuyển động trượt của vật (hình c). Điều đó chứng tỏ lực ma sát khi vật chuyển động nhỏ hơn lực ma sát nghỉ cực đại.
\end{itemize}	
\end{note}
\subsubsection{Lực ma sát trượt}
Lực ma sát trượt là lực ma sát cản trở vật trượt trên bề mặt tiếp xúc.
\begin{align*}
\boder{F_{ms}=\mu N}
\end{align*}

\immini{\begin{itemize}
\item Trong đó:
\begin{itemize}
\item N là áp lực lên mặt tiếp xúc.
\item $\mu$ là hệ số ma sát trượt, chỉ phụ thuộc vào vật liệu và tình trạng của hai mặt tiếp xúc.
\end{itemize}	
\item Lực ma sát trượt có phương song song với bề mặt tiếp xúc.
\item Lực ma sát trượt có chiều ngược chiều chuyển động của vật.
\item Lực ma sát trượt \textit{gần như} không phụ thuộc vào diện tích tiếp xúc và tốc độ chuyển động của vật.

\end{itemize}}{\begin{tabular}{|c|c|}

\begin{note}
Nếu vật trượt trên mặt phẳng ngang:
\begin{align*}
\boder{N=P \Rightarrow F_{ms}=\mu P =\mu mg}
\end{align*}	
\end{note}
\immini{
\subsubsection{Ứng dụng của lực ma sát}\begin{itemize}
\item Người di chuyển trên đường nhờ lực ma sát.
\item Vận động viên thể dục dụng cụ xoa phấn vào lòng bàn tay trước khi nâng tạ để tăng ma sát.
\item Một số trường hợp lực ma sát có tác dụng cản trở chuyển động, nhưng cũng có trường hợp lực ma sát thúc đẩy chuyển động. 
\item Trong đời sống và trong kĩ thuật, để giảm ma sát người ta thay vật trượt bằng vật lăn (hình cầu hoặc hình trụ).
\end{itemize}
