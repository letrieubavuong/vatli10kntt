\subsubsection{ĐỊNH LUẬT 2 NEWTON}
\begin{itemize}
\item Mối liên hệ giữa ba đại lượng: \textit{gia tốc, lực} và \textit{khối lượng} đã được Newton khái quát trong một phương trình vector đơn giản gọi là định luật 2 Newton:\\
\textit{Gia tốc của một vật cùng hướng với lực tác dụng lên vật. Độ lớn của gia tốc tỉ lệ thuận với độ lớn của lực và tỉ lệ nghịch với khối lượng của vật}
\begin{align*}
\boder{\vec{a}=\dfrac{\vec F}{m}\Rightarrow \vec F=m\vec a}
\end{align*}
\item Trong trường hợp vật chịu nhiều lực tác dụng $\vec F_1,\ \vec F_2,\ \vec F_3,...$ thì $\vec F$ là hợp lực của các lực đó:
\begin{align*}
\vec F=\vec F_1+\vec F_2+\vec F_3+...
\end{align*}
\end{itemize}

\subsubsection{KHỐI LƯỢNG VÀ MỨC QUÁN TÍNH}
\begin{itemize}
\item Theo định luật 2 Newton, nếu có nhiều vật khác nhau lần lượt chịu tác dụng của cùng một lực không đổi, thì vật nào có khối lượng lớn hơn sẽ có gia tốc nhỏ hơn. Vật nào có khối lượng càng lớn thì càng khó thay đổi vận tốc, tức là càng có \textit{mức quán tính} lớn hơn. Từ đó ta có thể nói: \textit{Khối lượng của vật là đại lượng đặc trưng cho mức quán tính của vật}.
\item Điều đó cho phép ta so sánh được khối lượng của những vật làm bằng các chất khác nhau. Một xe chở cát và một xe chở gạo được coi là có khối lượng bằng nhau nếu dưới tác dụng của hợp lực như nhau, chúng có gia tốc như nhau.
\end{itemize}	
\begin{note}
\begin{itemize}
\item Khối lượng của định luật 2 Newton còn được gọi là khối lượng quán tính.
\item Khối lượng mà đại lượng vô hướng, dương, không đổi với mỗi vật và có tính chất cộng được.
\end{itemize}	
\end{note}
